CALL [initmem]

CALL [terminate]

initmem:
LD DA 256
SET_PAGE_STACK
LD DA 512
SET_PAGE_HEAP
PAGE_INIT_MEM
RET

terminate:
LD A 0
INT 0
RET

prtstr:
LODS
LD A 1
INT 1
RET

readstr:
LD A 5
INT 1
POPS
RET
